#题目描述
有 n 个人围坐成一圈，其编号为从 1 到 n 的递增数列，每个人有一个正整数密码。先选定一个任意正整数 m，并从 1 号开始报 1，2号报 2，以此类推，报到 m 时停止，该人出局，并把他的密码作为新的 m 值。重复该过程，直到所有人出局，求出局人编号顺序。

#代码
`
//
// Created by Kevin on 2016/10/8.
//

#include

#include

typedef struct node {
int number;
int password;
struct node *next;
} node;

node createCircularLinkedList(int n) {
node temp = NULL;
node head = (node ) malloc(sizeof(node));
temp = head;
head->number = 1;
for (int i = 2; i <= n; i++) {
temp->next = (node *) malloc(sizeof(node));
temp->next->number = i;
temp = temp->next;
temp->next = NULL;
}
temp->next = head;
return temp;
}//返回链表尾指针

int initCircularLinkedList(node rear) {
if (NULL == rear) // If a blank linked list.
return 0;
node temp = rear;
int password;
printf(“Please input password:”);
do {
scanf(“%d”, &password);
temp->next->password = password;
temp = temp->next;
} while (temp != rear);
return 1;
}

//找到被删除结点的前驱
node getDeletedNodePioneer(node rear, int m, int n) {
if (NULL == rear)
return NULL;
node *temp = rear;
m = (m % n == 0 ? n : (m % n));//如果模为0，循环n次
for (int i = 1; i < m; i++) {
temp = temp->next;
}
return temp;

}

node deleteCircularLinkedListNode(node pioneer) {
if (NULL == pioneer)
return NULL;
node temp = pioneer, p;
p = temp->next;
printf(“%d “, p->number);
temp->next = temp->next->next;
free(p);
return temp;
}

int main() {
int n = 0, m = 0, tempm = 0;
node rear = NULL, p = NULL;
printf(“Please input n:”);
scanf(“%d”, &n);
rear = createCircularLinkedList(n);//新建链表
initCircularLinkedList(rear);//初始化
printf(“Please input m:”);
scanf(“%d”, &m);

while (n > 0) {
    p = getDeletedNodePioneer(rear, m, n);
    tempm = p->next->password;//得到新的m值
    rear = deleteCircularLinkedListNode(p);
    m = tempm;
    n--;
}
return 0;

}
